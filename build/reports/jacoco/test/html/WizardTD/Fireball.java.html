public class Fireball {
    private int x;
    private int y;
    private boolean isHit;
    private Monster monster;
    private Tower tower;

    public Fireball(Tower tower) {
        this.tower=tower;
        this.x=tower.getTowerX()+15;
        this.y=tower.getTowerY()+15;
        this.isHit=false;
    }
    public Tower getTower(){
        return this.tower;
    }
    public int getX(){
        return this.x;
    }
    public int getY(){
        return this.y;
    }
    public void updateX(int x){
        this.x=x;
    }
    public void updateY(int y){
        this.y=y;
    }
    public void setMonster(Monster monster){
        this.monster=monster;
    }
    public Monster getMonster(){
        return this.monster;
    }
    public int getTargetX(){
        return this.monster.getMonsterX();
    }
    public int getTargetY(){
        return this.monster.getMonsterY();
    }
    public void track(int targetx,int targety){
        float dx = targetx - x;
        float dy = targety - y;
        
        // 璁＄畻鍒扮洰鏍囩殑鎬昏窛绂�
        float distance=PApplet.dist(x, y, targetx, targety);
        // 璁＄畻鍗曚綅鍚戦噺
        float unitX = dx / distance;
        float unitY = dy / distance;
        
        // 璁＄畻姣忔搴旇绉诲姩鐨勮窛绂伙紙杩欓噷鏄�5鍍忕礌锛�
        float moveX = unitX * 5;
        float moveY = unitY * 5;
        
        // 濡傛灉璺濈澶т簬5锛屾甯哥Щ鍔紱鍚﹀垯锛岀洿鎺ュ埌杈剧洰鏍�
        if (distance > 5) {
            x += moveX;
            y += moveY;
        } else {
            x = targetx;
            y = targety;
            isHit = true;
        }
    }
    public boolean isHit(){
        return this.isHit;
    }

}
